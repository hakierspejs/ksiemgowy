135def gen_unseen_mbank_emails(database, mail): 

136 """Connects to imap_server using login and password from the arguments, 

137 then yields a pair (mail_id_as_str, email_as_eml_string) for each of 

138 e-mails coming from mBank.""" 

139 mail.select("inbox") 

140 _, data = mail.search(None, IMAP_FILTER) 

141 mail_ids = data[0] 

142 id_list = mail_ids.split() 

143 for mail_id in reversed(id_list): 

144 _, data = mail.fetch(mail_id, "(RFC822)") 

145 for mail_number, response_part in enumerate(data): 

146 if not isinstance(response_part, tuple): 

147 continue 

148 msg = email.message_from_string(response_part[1].decode()) 

149 mail_key = f'{msg["Date"]}_{mail_number}' 

150 if database.was_imap_id_already_handled(mail_key): 

151 continue 

152 LOGGER.info("Handling e-mail id: %r", mail_id) 

153 yield msg 

154 database.mark_imap_id_already_handled(mail_key)
