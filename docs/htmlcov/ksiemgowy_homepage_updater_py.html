40def upload_value_to_graphite(host, metric, value): 

41 """Uploads a single metric to a Graphite server.""" 

42 sock = socket.socket() 

43 try: 

44 sock.connect(host) 

45 now = int(time.time()) 

46 buf = f"{metric} {value} {now}\n".encode() 

47 LOGGER.info("Sending %r to %r", buf, host) 

48 sock.send(buf) 

49 sock.close() 

50 except (ConnectionRefusedError, socket.timeout, TimeoutError) as err: 

51 LOGGER.exception(err) 

52 time.sleep(3.0)

69def get_remote_state_dues(remote_state_path): 

70 """Reads remote state from a specified file, then de-serializes it 

71 and returns in a format that's similar to the one generated by 

72 Ksiemgowy.""" 

73 try: 

74 with open(remote_state_path, encoding="utf8") as remote_state_file:

81def ssh_agent_import_key_and_build_env_and_setup_git(deploy_key_path): 

82 """Creates an SSH agent session and adds a specified key to it. Returns 

83 environment variables needed to use the SSH agent.""" 

84 env = {} 

85 for line in subprocess.check_output(["ssh-agent", "-c"]).split(b"\n"): 

86 split = line.decode().split() 

87 if ( 

88 len(split) == 3 

89 and split[0] == "setenv" 

90 and split[-1].endswith(";") 

91 ): 

92 env[split[1]] = split[2].rstrip(";") 

93 subprocess.check_call(["ssh-add", deploy_key_path], env=env) 

94 subprocess.check_call(["ssh-add", "-l"], env=env) 

95 return env

98def set_up_git_identity(username, email, cwd):

105@contextlib.contextmanager 

106def git_cloned(deploy_key_path): 

107 """Sets up SSH authorization to a Git repository using SSH deploy keys, 

108 then clones the repository and sets up git identify for it. Acts as a 

109 context manager - once it exits, the source tree gets deleted. 

110 

111 Yields enviroment variables needed to be able to commit and push to git.""" 

112 cwd = "homepage" 

113 try: 

114 env = ssh_agent_import_key_and_build_env_and_setup_git(deploy_key_path) 

115 git_url = f"git@github.com:{HOMEPAGE_REPO}.git" 

116 env["GIT_SSH_COMMAND"] = " ".join( 

117 [ 

118 "ssh", 

119 "-o", 

120 "StrictHostKeyChecking=no", 

121 "-o", 

122 "UserKnownHostsFile=/dev/null", 

123 ] 

124 ) 

125 subprocess.check_call(["git", "clone", git_url, cwd], env=env) 

126 set_up_git_identity("ksiemgowy", "d33tah+ksiemgowy@gmail.com", cwd)

146def do_states_differ(remote_state, current_report): 

147 """Returns true if current report differs from the remote one. Does not 

148 compare keys which are not in current report.""" 

149 for k in current_report: 

150 if current_report.get(k) != remote_state.get(k): 

151 return True 

152 return False

155def is_newer(remote_state, current_report): 

156 """Compares remote state and current report, using time criteria.""" 

157 local_modified = datetime.datetime.strptime( 

158 current_report["dues_last_updated"], "%d-%m-%Y" 

159 ) 

160 remote_modified = datetime.datetime.strptime( 

161 remote_state["dues_last_updated"], "%d-%m-%Y" 

162 ) 

163 return local_modified > remote_modified

166def maybe_update_dues(database, git_env): 

167 """Generates the current report, retrieves the one that's accessible online 

168 and if the current one is later, updates the remote state.""" 

169 now = datetime.datetime.now() 

170 current_report = ksiemgowy.current_report_builder.get_current_report( 

171 now, database.list_expenses(), database.list_positive_transfers() 

172 ) 

173 upload_to_graphite(GRAPHITE_HOST, current_report) 

174 remote_state_path = f"homepage/{DUES_FILE_PATH}" 

175 remote_state = get_remote_state_dues(remote_state_path) 

176 has_changed = do_states_differ(remote_state, current_report) 

177 if has_changed and is_newer(remote_state, current_report): 

178 LOGGER.info("maybe_update_dues: updating dues") 

179 remote_state.update(current_report) 

180 update_git_remote_state(remote_state_path, remote_state, git_env) 

181 LOGGER.info("maybe_update_dues: done")
