107 """Creates an SSH agent session and adds a specified key to it. Returns 

108 environment variables needed to use the SSH agent.""" 

109 env = {} 

110 for line in subprocess.check_output(["ssh-agent", "-c"]).split(b"\n"): 

111 split = line.decode().split() 

112 if ( 

113 len(split) == 3 

114 and split[0] == "setenv" 

115 and split[-1].endswith(";") 

116 ): 

117 env[split[1]] = split[2].rstrip(";") 

118 subprocess.check_call(["ssh-add", deploy_key_path], env=env)
