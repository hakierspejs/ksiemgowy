214def build_monthly_final_balance( 

215 months: Set[str], 

216 monthly_income: Dict[str, Dict[str, float]], 

217 monthly_expenses: Dict[str, Dict[str, float]], 

218) -> Tuple[Dict[str, Dict[str, float]], float]: 

219 """Calculates monthly final balances, given all of the actions - an amount 

220 that specifies whether we accumulated more than we spent, or otherwise.""" 

221 balance_so_far = 0.0 

222 monthly_final_balance: Dict[str, Dict[str, float]] = {} 

223 for month in sorted(months): 

224 _monthly_income = sum(monthly_income.get(month, {}).values()) 

225 _monthly_expenses = sum(monthly_expenses.get(month, {}).values()) 

226 balance_so_far += _monthly_income - _monthly_expenses 

227 monthly_final_balance.setdefault(month, {}).setdefault("Suma", 0) 

228 monthly_final_balance[month]["Suma"] = balance_so_far 

229 return monthly_final_balance, balance_so_far
